% Slide 4 - Fixed-beta problem and research question
% =============================================================
\begin{frame}{Fixed-$\beta$ Problem and Research Question}
  \vspace{1em}
  \begin{itemize}\small
    \setlength{\itemsep}{0.6em}
    \item Same $\beta$ for every state -- but useful exploration varies per state.
    \item Static throughout training -- exploration need shifts as the agent learns.
    \item Decoupled from task signal -- intrinsic bonus fires whether or not it helps extrinsic reward.
  \end{itemize}

  \vspace{1em}
  \begin{block}{Research question}
    \footnotesize\itshape Can we adapt the intrinsic-reward coefficient automatically during training so that exploration better supports extrinsic-reward learning?
  \end{block}
\end{frame}

% =============================================================
% Slide 5 - Proposed Method: CARE
% =============================================================
\begin{frame}{Proposed Method: CARE}
  \vspace{1.5em}
  \centering
  \begin{tabular}{rc}
    \textbf{Before:} & $\displaystyle r_t = r_t^{ext} + \beta\,r_t^{int}$ \\[1em]
    \textbf{CARE:}   & $\displaystyle r_t = r_t^{ext} + \boxed{\beta_\psi(s_t)}\,r_t^{int}$ \\
  \end{tabular}

  \vspace{1.5em}
  \begin{block}{Main idea}
    \small Raise intrinsic reward when exploration helps future extrinsic reward; reduce it when distracting.
  \end{block}
\end{frame}

% =============================================================
% Slide 6 - CARE Training Objective
% =============================================================
\begin{frame}{CARE Training Objective}
  \vspace{0.6em}
  \begin{itemize}\small
    \setlength{\itemsep}{0.3em}
    \item \textbf{\color{uetblue}Correlation} -- align intrinsic reward with future extrinsic advantage.
    \item \textbf{\color{uetorange}Regularizer} -- keep $\beta_\psi$ near $\beta_0$ in log-space.
  \end{itemize}

  \vspace{1em}
  \[
    \mathcal{L}_\beta(\psi)
    \;=\;
    \underbrace{\fcolorbox{uetblue}{uetblue!10}{$\displaystyle -\,\mathbb{E}\bigl[\,\hat{I}_z \cdot \hat{A}^E_z \,\bigr]\,$}}_{\text{\footnotesize\color{uetblue}\bfseries correlation}}
    \;+\;
    \underbrace{\fcolorbox{uetorange}{uetorange!15}{$\displaystyle \,\lambda_\text{reg}\,\bigl(\log \beta_\psi(s) - \log \beta_0\bigr)^2\,$}}_{\text{\footnotesize\color{uetorange}\bfseries log-space regularizer}}
  \]
\end{frame}

% Slide 11 - Limitations

\begin{frame}{Limitations}
  \vspace{1em}
  \begin{itemize}\small
    \setlength{\itemsep}{0.9em}
    \item MiniGrid only (discrete, low-dim).
    \item Extra network + training objective.
    \item Sensitive to $\beta_0$, $\lambda_\text{reg}$, lr$_\psi$.
    \item Future: continuous control, image-based.
  \end{itemize}
\end{frame}

% =============================================================
% Slide 12 - Conclusion
% =============================================================
\begin{frame}{Conclusion}
  \vspace{1em}
  \begin{itemize}\small
    \setlength{\itemsep}{0.8em}
    \item Sparse rewards $\Rightarrow$ hard exploration.
    \item Fixed $\beta$ = one-size-fits-all.
    \item \textbf{CARE}: adaptive $\beta_\psi(s)$ per state.
  \end{itemize}

\begin{frame}{[Backup] CARE Pseudocode (Algorithm 1)}
  \footnotesize
  \begin{enumerate}
    \item Initialize $\pi_\theta$, $V_\phi$, Beta Network $\beta_\psi$, intrinsic module $m$.
    \item \textbf{Repeat} for $T$ steps:
    \begin{enumerate}\footnotesize
      \item Collect rollout $\{(s_t,a_t,r^E_t,s_{t+1})\}$ with $\pi_\theta$.
      \item Compute $R^{I,m}_t$ from module $m$; standardize to $I^+_t$.
      \item Compute $\beta_\psi(s_t)$ and shaped reward $\bar{r}_t = r^E_t + \beta_\psi(s_t)\cdot I^+_t$.
      \item Compute raw extrinsic GAE advantage $\hat{A}^E_t$ and shaped advantage $\hat{A}^{\text{shaped}}_t$.
      \item Update $\pi_\theta$, $V_\phi$ with PPO ($K{=}80$ epochs, normalized $\hat{A}$).
      \item \textbf{One} Adam step on $\psi$: $\mathcal{L}_\beta = -\mathbb{E}[\hat{I}_z\,\hat{A}^E_z] + \lambda_\text{reg}(\log\beta_\psi - \log\beta_0)^2$.
      \item Clip $\beta_\psi$ to $[\beta_\text{min},\beta_\text{max}]$.
    \end{enumerate}
  \end{enumerate}
\end{frame}

\begin{frame}{[Backup] Key Hyperparameters}
  \footnotesize
  \begin{tabular}{ll}
    \toprule
    \textbf{PPO} & $K=80$, $\epsilon_\text{clip}=0.2$, $\gamma=0.99$, $\lambda_\text{GAE}=0.95$ \\
                 & lr$_\text{actor}=3{\times}10^{-4}$, lr$_\text{critic}=10^{-3}$, rollout $T=4{,}000$ \\
    \midrule
    \textbf{CARE} & $\beta_\text{min}=10^{-4}$, $\beta_\text{max}=5{\times}10^{-2}$ \\
                  & $\beta_0=\sqrt{\beta_\text{min}\beta_\text{max}}\approx 2.24{\times}10^{-3}$ \\
                  & $\lambda_\text{reg}=10^{-3}$, lr$_\psi=5{\times}10^{-4}$, weight decay $10^{-6}$ \\
                  & one $\psi$-step per PPO update, grad-clip L2 $=1.0$, no delayed target copy \\
    \midrule
    \textbf{Fixed-$\beta$ sweep} & $\beta \in \{0.0005,\,0.001,\,0.005,\,0.01,\,0.05\}$ \\
    \midrule
    \textbf{Seeds} & $\{1,2,3,4,5\}$, $10^6$ env steps per run \\
    \bottomrule
  \end{tabular}

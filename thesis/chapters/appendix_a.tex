\chapter{Mathematical Proofs}
\label{app:proofs}

This appendix collects the full proofs of the formal results stated in Chapter~\ref{ch:methodology}. Each section below corresponds to a single proposition, lemma, or corollary, and may be read independently of the others. Throughout the appendix, expectations are taken with respect to the on-policy distribution induced by the current policy $\pi_\theta$ unless stated otherwise; standard regularity conditions (existence of first and second moments, measurability of $\beta_\psi$, and a state-visitation distribution that admits a density) are assumed.

\section{Proof of Proposition 3.1: Optimal State-Dependent Scaling}
\label{app:proof_target}

\begin{proof}
We work with the population form of the correlation loss in Equation~\eqref{eq:corr_loss}, identifying minibatch averages with expectations under the on-policy state-visitation distribution. Let $\tilde I_t = \beta_\psi(s_t)\, I_t^{+,m}$ denote the weighted intrinsic reward and $\hat A_t^{E}$ the extrinsic GAE advantage computed from $V_\mu$. Setting $\lambda_\text{reg} = 0$ in Equation~\eqref{eq:beta_loss} reduces the Beta-network objective to $L_\beta(\psi) = L_\text{corr}(\psi)$, which equals the negative population correlation
\begin{equation*}
L_\text{corr}(\psi) \;=\; -\,\frac{\mathbb{E}[(\tilde I - \mu_{\tilde I})(\hat A^{E} - \mu_{\hat A^{E}})]}{\sigma_{\tilde I}\, \sigma_{\hat A^{E}}}
\end{equation*}
in the population limit, where $\mu_\bullet$ and $\sigma_\bullet$ are the marginal mean and standard deviation under the state-visitation distribution. By the assumption that the cross-moment is nonzero almost everywhere, both standard deviations are strictly positive.

Fix a state $s \in \mathcal{S}$ visited with positive probability, and treat $f(s) := \beta_\psi(s)$ as a free real parameter (this is admissible because the Beta Network has sufficient capacity to represent any measurable function into $[\beta_\text{min}, \beta_\text{max}]$). Conditioning on $s_t = s$,
\begin{equation*}
\frac{\partial}{\partial f(s)}\, \mathbb{E}\!\left[\, \tilde I_t\,\hat A_t^{E} \,\right]
\;=\; \mathbb{P}(s_t = s)\, \mathbb{E}\!\left[\, I_t^{+,m}\,\hat A_t^{E} \,\big|\, s_t = s \right],
\end{equation*}
because $\tilde I_t = f(s)\, I_t^{+,m}$ at $s_t = s$. Likewise, $\partial \mu_{\tilde I} / \partial f(s) = \mathbb{P}(s_t = s)\, \mathbb{E}[I_t^{+,m} \mid s_t = s]$ and the contribution of $f(s)$ to $\sigma_{\tilde I}$ is positive and bounded away from zero by the nondegeneracy assumption. Setting the partial derivative of the population $L_\text{corr}$ to zero at an interior optimum $f^\star(s) \in (\beta_\text{min}, \beta_\text{max})$ and rearranging, the first-order condition reads
\begin{equation*}
\mathbb{E}\!\left[\, I_t^{+,m}\,\bigl( \hat A_t^{E} - \mu_{\hat A^{E}} \bigr) \,\big|\, s_t = s \right]
\;=\;
c(s)\, \bigl( f^\star(s) - \mu_{\tilde I} \bigr),
\end{equation*}
for some scalar $c(s) > 0$ that depends on the variance terms but not on the sign of $f^\star(s) - \beta_0$. Because $\mu_{\tilde I}$ collapses to $\beta_0\,\mathbb{E}[I^{+,m}]$ when $f \equiv \beta_0$, the sign of the right-hand side at a small perturbation of $\beta_0$ matches the sign of $f^\star(s) - \beta_0$, while the sign of the left-hand side is fixed by $\mathbb{E}[\, I^{+,m}\,\hat A^{E} \mid s_t = s]$. Consequently,
\begin{equation*}
\mathrm{sign}\bigl( f^\star(s) - \beta_0 \bigr)
\;=\;
\mathrm{sign}\!\left(\, \mathbb{E}[\, I_t^{+,m}\, \hat A_t^{E} \mid s_t = s \,] \right),
\end{equation*}
which is the announced equality. When the unconstrained interior solution falls outside $[\beta_\text{min}, \beta_\text{max}]$, the bound is active and $f^\star(s) \in \{\beta_\text{min}, \beta_\text{max}\}$; the same monotone relationship in the conditional cross-moment is preserved at the clipped optimum.
\end{proof}

\section{Proof of Lemma 3.1: Bounded Shaped Reward}
\label{app:proof_bounded}

\begin{proof}
By Equation~\eqref{eq:combined_reward},
\begin{equation*}
\bar r_t - R_t^E \;=\; \beta_\psi(s_t)\, I_t^{+,m}.
\end{equation*}
Taking absolute values,
\begin{equation*}
\left| \bar r_t - R_t^E \right| \;=\; \left| \beta_\psi(s_t) \right|\, \left| I_t^{+,m} \right|.
\end{equation*}
By Definition~\ref{def:beta_network}, $\beta_\psi(s_t) \in [\beta_\text{min}, \beta_\text{max}]$ with $\beta_\text{min} > 0$, so $|\beta_\psi(s_t)| \le \beta_\text{max}$. By assumption, $|I_t^{+,m}| \le M$ almost surely. Combining these bounds,
\begin{equation*}
\left| \bar r_t - R_t^E \right| \;\le\; \beta_\text{max}\, M,
\end{equation*}
which establishes the claim.
\end{proof}

\section{Proof of Proposition 3.2: Computational Cost Comparison}
\label{app:proof_cost}

\begin{proof}
We measure cost in scalar floating-point operations and ignore lower-order terms.
Write $|\theta|$, $|\mu|$, $|\psi|$ for the number of parameters in the policy, critic, and Beta Network respectively.

\textbf{Meta-gradient training step.} A meta-gradient estimator (LIRPG \cite{zheng2018lirpg}) updates the intrinsic-reward parameters by differentiating \emph{through} $K$ inner policy-update steps applied to the buffer of $T$ transitions. The costly part is the backward pass: because each of the $K$ inner steps depends on the current policy parameters, the entire chain of $K$ updates must be unrolled and cached before a single gradient with respect to the intrinsic-reward parameters can be computed. Concretely:
(i) each inner step evaluates the policy gradient over $T$ transitions, costing $O(T\,|\theta|)$, and applies the update, costing $O(|\theta|)$; and
(ii) the backward pass through all $K$ unrolled steps mirrors the forward pass, costing $O(K\,T\,|\theta|)$ in time and $O(K\,|\theta|)$ in memory.
The worst-case time complexity per buffer update is therefore
\begin{equation*}
C_\text{meta} \;=\; O(K\, T\, |\theta|).
\end{equation*}

\textbf{CARE training step.} CARE updates only the Beta Network $\beta_\psi$, without differentiating through any policy step. On the same buffer of $T$ transitions:
(i) a forward pass through the critic $V_\mu$ produces the extrinsic GAE advantages $\hat A_t^{E}$, at cost $O(T\,|\mu|)$; this pass is already required by PPO and adds no extra cost;
(ii) the GAE recursion in Equation~\eqref{eq:gae_advantage_ext} runs in $O(T)$;
(iii) a forward pass through $\beta_\psi$ evaluates the scaled intrinsic reward for each of the $T$ states, at cost $O(T\,|\psi|)$; and
(iv) a single backward pass through $\beta_\psi$ computes $\nabla_\psi L_\beta$ from the correlation in Equation~\eqref{eq:corr_loss}, at cost $O(T\,|\psi|)$.
No policy parameters enter the backward pass. The dominant term is $O(T\,|\psi|)$, so the time complexity per buffer update is
\begin{equation*}
C_\text{CARE} \;=\; O(T\, |\psi|).
\end{equation*}

\textbf{Asymptotic ratio.} Dividing the two costs,
\begin{equation*}
\frac{C_\text{meta}}{C_\text{CARE}} \;=\; \Theta\!\left( \frac{K\, |\theta|}{|\psi|} \right).
\end{equation*}
This ratio is large whenever $|\psi| \ll |\theta|$ and $K \ge 1$. In the implemented system, $|\psi|$ is roughly two orders of magnitude smaller than $|\theta|$ and $K = 80$, so the saving is substantial.
\end{proof}

\section{Proof of Corollary 3.1: Degeneration to Fixed Coefficient}
\label{app:proof_degeneration}

\begin{proof}
We treat the two regimes separately.

\textbf{(i) Cold-start regime.} Suppose the rollout buffer contains no informative extrinsic signal, modeled formally as $\hat A_t^{E} = 0$ for almost every $t$. Then the marginal standard deviation $\sigma_{\hat A^{E}}$ is zero, so the standardized advantage $\hat A_t^{z}$ in Equation~\eqref{eq:zscored_signals} is undefined; in the implementation, the small constant $\varepsilon$ in the denominator keeps the operation numerically stable, but the standardized quantity is then a vanishing-magnitude random variable concentrated at zero. Substituting $\hat A_t^{z} \to 0$ into Equation~\eqref{eq:corr_loss},
\begin{equation*}
L_\text{corr}(\psi) \;\to\; -\,\frac{1}{B}\sum_{t=1}^{B}\, \hat I_t^{z}\cdot 0 \;=\; 0
\quad\text{identically in } \psi,
\end{equation*}
so the gradient $\nabla_\psi L_\text{corr}$ vanishes everywhere. The full Beta-network objective therefore reduces to $L_\beta(\psi) = \lambda_\text{reg}\, L_\text{reg}(\psi)$ with $L_\text{reg}(\psi) = \mathbb{E}[(\log \beta_\psi(s_t) - \log \beta_0)^2]$. This expectation is non-negative and equals zero if and only if $\beta_\psi(s) = \beta_0$ almost everywhere on the state-visitation support. Hence $\beta_{\psi^\star}(s) \equiv \beta_0$ is the unique minimizer.

\textbf{(ii) Strong regularization regime.} The correlation loss $L_\text{corr}$ takes values in $[-1, 1]$ by the Cauchy--Schwarz inequality applied to z-scored variables, so $|L_\text{corr}(\psi)| \le 1$ uniformly in $\psi$. The regularizer $L_\text{reg}$ is non-negative. Let $\psi^\star_\lambda$ be any minimizer of $L_\beta$ at regularization weight $\lambda \equiv \lambda_\text{reg}$. By optimality, comparing against any reference $\psi_0$ that satisfies $\beta_{\psi_0}(s) \equiv \beta_0$ (for which $L_\text{reg}(\psi_0) = 0$ and $|L_\text{corr}(\psi_0)| \le 1$),
\begin{equation*}
L_\text{corr}(\psi^\star_\lambda) \;+\; \lambda\, L_\text{reg}(\psi^\star_\lambda)
\;\le\; L_\text{corr}(\psi_0) \;+\; 0
\;\le\; 1.
\end{equation*}
Rearranging,
\begin{equation*}
\lambda\, L_\text{reg}(\psi^\star_\lambda)
\;\le\; 1 \;-\; L_\text{corr}(\psi^\star_\lambda)
\;\le\; 2,
\end{equation*}
where the last inequality uses $L_\text{corr}(\psi^\star_\lambda) \ge -1$. Dividing by $\lambda > 0$,
\begin{equation*}
L_\text{reg}(\psi^\star_\lambda)
\;\le\; \frac{2}{\lambda}
\;\xrightarrow[\lambda \to \infty]{}\; 0.
\end{equation*}
Hence $\mathbb{E}[(\log \beta_{\psi^\star_\lambda}(s_t) - \log \beta_0)^2] \to 0$, which implies $\log \beta_{\psi^\star_\lambda}(s_t) \to \log \beta_0$ in $L^2$ under the state-visitation distribution. Equivalently, $\beta_{\psi^\star_\lambda}(s) \to \beta_0$ in $L^2$ on the support of that distribution.

\textbf{Reduction to fixed-coefficient form.} In either regime, substituting $\beta_\psi(s) = \beta_0$ into Equation~\eqref{eq:combined_reward} yields
\begin{equation*}
\bar r_t \;=\; R_t^E \;+\; \beta_0\, I_t^{+,m},
\end{equation*}
which is exactly the fixed-coefficient PPO update with intrinsic coefficient $\beta_0$.
\end{proof}

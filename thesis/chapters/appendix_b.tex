\chapter{Implementation Details}
\label{app:implementation}

This appendix records the network architectures, hyperparameters, and training schedule used in the experiments reported in Chapter~\ref{ch:methodology} and Chapter 4. All values reflect the canonical settings adopted across the six benchmark MiniGrid environments and are extracted from the project source code.

\section{Network Architectures}
\label{app:architectures}

\begin{table}[H]
\centering
\caption{Beta Network $\beta_\psi$ architecture.}
\label{tab:beta_arch}
\begin{tabular}{|l|l|}
\hline
\textbf{Component} & \textbf{Specification} \\
\hline
Input & state $s_t$ flattened to $\mathbb{R}^{d_s}$ \\
\hline
Encoder layer 1 & Linear($d_s \to 256$), Tanh \\
\hline
Encoder layer 2 & Linear($256 \to 256$), Tanh \\
\hline
Head layer 1 & Linear($256 \to 128$), Tanh \\
\hline
Head output & Linear($128 \to 1$) producing $\log \beta_\psi(s_t)$ \\
\hline
Output bound & $\beta_\psi(s_t) = \exp\!\bigl( \mathrm{clamp}(\log \beta_\psi,\, \log \beta_\text{min},\, \log \beta_\text{max}) \bigr)$ \\
\hline
$[\beta_\text{min}, \beta_\text{max}]$ & $[0.1,\, 2.0]$ \\
\hline
Encoder weight init & $\mathcal{N}(0,\, 1/\sqrt{d_\text{in}})$ \\
\hline
Output bias init & $\log(1.0) = 0$ (so $\beta_\psi \approx 1$ at initialization) \\
\hline
\end{tabular}
\end{table}

\begin{table}[H]
\centering
\caption{Intrinsic Curiosity Module (ICM) architecture.}
\label{tab:icm_arch}
\begin{tabular}{|l|l|}
\hline
\textbf{Component} & \textbf{Specification} \\
\hline
Encoder $\phi$ layer 1 & Linear($d_s \to 256$), Tanh \\
\hline
Encoder $\phi$ layer 2 & Linear($256 \to 256$), Tanh \\
\hline
Latent dimension & $256$ \\
\hline
Forward model $f^F_\eta$ & Linear($256 + |\mathcal{A}| \to 256$), Tanh, Linear($256 \to 256$) \\
\hline
Inverse model $f^I_\eta$ & Linear($512 \to 256$), Tanh, Linear($256 \to |\mathcal{A}|$) \\
\hline
Forward weight $\alpha_F$ in $L_\text{ICM}$ & $0.2$ \\
\hline
Inverse weight $\alpha_I$ in $L_\text{ICM}$ & $0.8$ \\
\hline
Action representation in forward model & one-hot $\in \mathbb{R}^{|\mathcal{A}|}$ \\
\hline
Encoder gradient flow & inverse loss only (forward loss detached) \\
\hline
\end{tabular}
\end{table}

\begin{table}[H]
\centering
\caption{RIDE architecture. The encoder, forward model, and inverse model share the ICM design and training objective; the difference is the intrinsic reward formula in Equation~\eqref{eq:ride_raw}, which uses representation impact divided by an episodic count rather than forward prediction error.}
\label{tab:ride_arch}
\begin{tabular}{|l|l|}
\hline
\textbf{Component} & \textbf{Specification} \\
\hline
Encoder $\phi$ layer 1 & Linear($d_s \to 256$), Tanh \\
\hline
Encoder $\phi$ layer 2 & Linear($256 \to 256$), Tanh \\
\hline
Latent dimension $d_y$ & $256$ \\
\hline
Forward model $f^F_\eta$ & Linear($256 + |\mathcal{A}| \to 256$), Tanh, Linear($256 \to 256$) \\
\hline
Inverse model $f^I_\eta$ & Linear($512 \to 256$), Tanh, Linear($256 \to |\mathcal{A}|$) \\
\hline
Episodic-count hash dim $d_h$ & $32$ \\
\hline
Episodic-count bin width $\Delta$ & $0.1$ \\
\hline
Episodic count reset & At every episode boundary (via \texttt{is\_terminals} buffer) \\
\hline
Forward/inverse weighting & $\alpha_F = 0.2$, $\alpha_I = 0.8$ (same as ICM) \\
\hline
Intrinsic signal & $I_t = \| \phi(s_{t+1}) - \phi(s_t) \|_2 / \sqrt{\max\{N_e(s_{t+1}), 1\}}$ \\
\hline
\end{tabular}
\end{table}
